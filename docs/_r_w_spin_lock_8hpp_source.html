173  void lock() {

238  void unlock_and_lock_upgrade() {
239  // need to do it in two steps here -- as the UPGRADED bit might be OR-ed
240  // at the same time when other threads are trying do try_lock_upgrade().
241  bits_.fetch_or(UPGRADED, std::memory_order_acquire);
242  bits_.fetch_add(-WRITER, std::memory_order_release);
243  }

246  bool try_lock() {
247  int32_t expect = 0;
248  return bits_.compare_exchange_strong(
249  expect, WRITER, std::memory_order_acq_rel);
250  }

261  bool try_lock_shared() {
262  // fetch_add is considerably (100%) faster than compare_exchange,
263  // so here we are optimizing for the common (lock success) case.
264  int32_t value = bits_.fetch_add(READER, std::memory_order_acquire);
265  if (value & (WRITER | UPGRADED)) {
266  bits_.fetch_add(-READER, std::memory_order_release);
267  return false;
268  }
269  return true;
270  }
271 
273  bool try_unlock_upgrade_and_lock() {
274  int32_t expect = UPGRADED;
275  return bits_.compare_exchange_strong(
276  expect, WRITER, std::memory_order_acq_rel);
277  }
278 
287  bool try_lock_upgrade() {
288  int32_t value = bits_.fetch_or(UPGRADED, std::memory_order_acquire);
289 
290  return ((value & (UPGRADED | WRITER)) == 0);
291  }
292 
294  int32_t bits() const {
295  return bits_.load(std::memory_order_acquire);
296  }
